:name DW.Factory:parts_circuit
:budget_cap max

; Assembler/craft chain: circuit, chip.

:const int MAX_TIER 10
:const double TGT_CIRCUIT 900.0
:const double TGT_CHIP 250.0

:global int dw_run
:global int dw_fac_gen
:global int dw_fac_state
:global int dw_fac_tier
:global double dw_fac_need
:global int dw_fac_err
:global double dw_fac_err_until
:global string dw_fac_task
:global string dw_fac_fail

:local int t
:local int my_gen
:local double need

wakeup()

start:
my_gen = dw_fac_gen
t = if(t < 1, 1, if(t > MAX_TIER, 1, t))

loop:
gotoif(end, my_gen != dw_fac_gen || dw_run == 0)
dw_fac_tier = t

gotoif(do_circuit, count("circuit", t) < TGT_CIRCUIT)
gotoif(do_chip, count("chip", t) < TGT_CHIP)
goto(next_tier)

next_tier:
t = if(t >= MAX_TIER, 1, t + 1)
waitframe()
goto(loop)

do_circuit:
dw_fac_state = 311
dw_fac_task = "Making circuits via Assembler"
dw_fac_fail = "Assembler busy"
gotoif(busy, active("assembler"))
need = min(150.0, TGT_CIRCUIT - count("circuit", t))
need = min(need, count("cable", t))
dw_fac_need = need
dw_fac_fail = "Need cable"
gotoif(no_input, need <= 0.)
dw_fac_err = 0
dw_fac_fail = ""
produce("cable", t, need, "assembler")
goto(next_tier)

do_chip:
dw_fac_state = 321
dw_fac_task = "Crafting chips"
need = min(60.0, TGT_CHIP - count("chip", t))
dw_fac_need = need
dw_fac_fail = "Need circuit materials"
gotoif(no_input, need <= 0.)
dw_fac_err = 0
dw_fac_fail = ""
craft("chip", t, need)
goto(next_tier)

busy:
dw_fac_err = 1
dw_fac_err_until = now() + 10000000.
goto(next_tier)

no_input:
dw_fac_err = 2
dw_fac_err_until = now() + 10000000.
goto(next_tier)

end:
